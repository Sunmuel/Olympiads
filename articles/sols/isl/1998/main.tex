\subsection{Problem 2.}
\begin{problem}{}{}
  Let $x,y$ and $z$ be positive real numbers such that $xyz=1$. Prove that


\[
 \frac{x^{3}}{(1 + y)(1 + z)}+\frac{y^{3}}{(1 + z)(1 + x)}+\frac{z^{3}}{(1 + x)(1 + y)}
 \geq \frac{3}{4}. 
\]
\end{problem}

\textcolor{color2}{\textbf{First solution}} By Cauchy-Schwarz,

\begin{align*}
  \left(\cyc \frac{x^3}{(1+y)(1+z)}\right)\left(\cyc (1+y)(1+z)\right) & \geq (x^{\frac{3}{2}}
  + y^{\frac{3}{2}} + z^{\frac{3}{2}})^2 \\ & = x^3+y^3+z^3+2((xy)^{\frac{3}{2}}+(yz)^{\frac{3}{2}}+(zx)^{\frac{3}{2}}).
\end{align*}
Therefore, it suffices to prove that
\begin{align*}
    & x^3+y^3+z^3+2((xy)^{\frac{3}{2}}+(yz)^{\frac{3}{2}}+(zx)^{\frac{3}{2}}) \geq \frac{3}{4}\left(3+2(x+y+z)+xy+yz+zx \right) \iff \\
    & 2 \sym x^3 +4 \sym (x^{11}y^{11}z^5)^{\frac{1}{9}} \geq \frac{3}{2} \sym abc+\sym(x^5y^2z^2)^{\frac{1}{3}}+\cyc(x^4y^4z)^{\frac{1}{3}}
\end{align*}
which is true, since
\begin{itemize}
  \item $\displaystyle \sym x^3 \geq \sym(x^5y^2z^2)^{\frac{1}{3}}$,
  \item $\displaystyle \sym x^3 \geq \cyc(x^4y^4z)^{\frac{1}{3}}$,
  \item $\displaystyle 4 \sym (x^{11}y^{11}z^5)^{\frac{1}{9}} \geq \frac{3}{2} \sym abc$,
\end{itemize}
are all true by the Muirhead's inequality

\textcolor{color2}{\textbf{Second solution}} By Titu's lemma,

\[
  \cyc \frac{x^4}{x(1+y)(1+z)} \geq \frac{(x^2+y^2+z^2)^2}{3xyz+2(xy+yz+za)+x+y+z} \geq \frac{(x^2+y^2+z^2)^2}{4(x^2+y^2+z^2)} \geq \frac{3}{4}
\]
